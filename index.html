Services / Controller interaction

Define a Service

    var app = angular.module('myapp', []);

    app.factory('User', function(){
      var user = { name: 'Bob', age: 30 };

      return {
        get: function() { return user; }
      };
    });

Use in a controller

    app.controller('ProfileController', function($scope, User) {
      $scope.user = User.get();
    });

Visible in a template

      <div ng-app="myapp">
        <div ng-controller="ProfileController">
          <div>{{user}}</div>
          <input type="text" ng-model="user.name" />
          <input type="text" ng-model="user.age" />
        </div>
      </div>
Working Example

Directives

      app.directive('profileEditor', function() {
        return {
          restrict: 'E',
          template: '<div>{{user}}</div> \
          <input type="text" ng-model="user.name" /> \
          <input type="text" ng-model="user.age" />'
        };
      });
    
      <div ng-app="myapp">
        <div ng-controller="ProfileController">
          <profile-editor></profile-editor>
        </div>
      </div>
Working Example

Filters

      app.filter('paddedNumber', function() {
          return function(input) {
            return String(input).indexOf('.') > -1 ? input : input + '.00' ;
          };
      });
    
      <div ng-app="myapp">
        <div ng-controller="ProfileController">
          <profile-editor></profile-editor>
          {{ user.age | paddedNumber }}
        </div>
      </div>
