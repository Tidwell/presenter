Organization - Module Structure

/*
	Tracks the number of times an element is clicked
*/
(function($,BI,undefined) {
	var $el;

	//tracks # of times $el is clicked
	var count = 0;

	function trackClick() {
		count++;
		return false; //stop prop & prev def
	}

	function init() {
		getEls();
		if (!$el) { return; }

		bindDOM();
	}

	function getEls() {
		$el = $('.selector');
	}

	function bindDOM() {
		$el.on('click','.selector',trackClick)
	}

	$(init);
}(jQuery,BI))
